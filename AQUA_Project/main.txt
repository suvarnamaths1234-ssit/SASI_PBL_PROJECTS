#include "LPC17xx.h"
#include <stdio.h>

// ============ PIN DEFINITIONS ============

// DHT11 pin
#define DHT11_PIN (1<<4)           // P0.4

// Ultrasonic pins
#define TRIG_PIN (1 << 5)          // P0.5
#define ECHO_PIN (1 << 6)          // P0.6

// Turbidity ADC Channel

// pH Sensor ADC Channel
#define PH_CHANNEL 1               // AD0.1 (P0.24)

// DO Sensor ADC Channel
#define DO_CHANNEL 2               // AD0.2 (P0.25)

// LCD Control pins - 4-bit mode
#define RS (1<<10)                 // P0.10
#define EN (1<<11)                 // P0.11

// LCD Data pins - 8-bit mode (P0.15 to P0.22)
#define LCD_DATA_MASK 0x007F8000   // P0.15-P0.22
#define LCD_DATA_SHIFT 15

// ============ CONSTANTS ============

// ============ DHT11 FUNCTIONS ============

uint8_t read_dht11(void) {
    uint8_t i, j;
    uint32_t timeout;
    
    for(i = 0; i < 5; i++) dht11_data[i] = 0;
    
    // Start signal
    LPC_GPIO0->FIODIR |= DHT11_PIN;
    LPC_GPIO0->FIOCLR = DHT11_PIN;
    delay_ms(18);
    LPC_GPIO0->FIOSET = DHT11_PIN;
    delay_us(30);
    LPC_GPIO0->FIODIR &= ~DHT11_PIN;
    
    // Wait for DHT11 response
    timeout = 0;
    while((LPC_GPIO0->FIOPIN & DHT11_PIN) && timeout++ < 1000);
    if(timeout >= 1000) return 0;
    
    timeout = 0;
    while(!(LPC_GPIO0->FIOPIN & DHT11_PIN) && timeout++ < 1000);
    if(timeout >= 1000) return 0;
    
    timeout = 0;
    while((LPC_GPIO0->FIOPIN & DHT11_PIN) && timeout++ < 1000);
    if(timeout >= 1000) return 0;
    
    // Read 40 bits
    for(i = 0; i < 5; i++) {
        for(j = 0; j < 8; j++) {
            timeout = 0;
            while(!(LPC_GPIO0->FIOPIN & DHT11_PIN) && timeout++ < 1000);
            if(timeout >= 1000) return 0;
            
            delay_us(30);
            if(LPC_GPIO0->FIOPIN & DHT11_PIN) {
                dht11_data[i] |= (1 << (7-j));
            }
            
            timeout = 0;
            while((LPC_GPIO0->FIOPIN & DHT11_PIN) && timeout++ < 1000);
            if(timeout >= 1000) return 0;
        }
    }
    
    // Verify checksum
    uint8_t checksum = dht11_data[0] + dht11_data[1] + dht11_data[2] + dht11_data[3];
    if(checksum != dht11_data[4]) return 0;
    
    humidity = (float)dht11_data[0] + ((float)dht11_data[1] / 10.0f);
    temperature = (float)dht11_data[2] + ((float)dht11_data[3] / 10.0f);
    
    return 1;
}

// ============ ULTRASONIC FUNCTIONS ============

float measureDistance(void) {
    unsigned int duration;
    float distance;
    unsigned int timeout;
    
    // Send trigger pulse
    LPC_GPIO0->FIOCLR = TRIG_PIN;
    for(volatile int i = 0; i < 50; i++);
    LPC_GPIO0->FIOSET = TRIG_PIN;
    for(volatile int i = 0; i < 250; i++);
    LPC_GPIO0->FIOCLR = TRIG_PIN;
    
    // Wait for echo high
    timeout = 0;
    while(!(LPC_GPIO0->FIOPIN & ECHO_PIN)) {
        timeout++;
        if(timeout > 300000) return -1;
    }
    
    Timer0_Start();
    
    // Wait for echo low
    timeout = 0;
    while(LPC_GPIO0->FIOPIN & ECHO_PIN) {
        timeout++;
        if(timeout > 300000) {
            Timer0_Stop();
            return -1;
        }
    }
    
    Timer0_Stop();
    duration = Timer0_GetValue();
    
    // Calculate distance
    distance = (duration * 0.0343) / 2.0;
    
    return distance;
}

void Timer0_Init(void) {
    LPC_SC->PCONP |= (1 << 1);
    LPC_TIM0->CTCR = 0x0;
    LPC_TIM0->PR = 24;
    LPC_TIM0->TCR = 0x02;
}

void Timer0_Start(void) {
    LPC_TIM0->TCR = 0x02;
    LPC_TIM0->TCR = 0x01;
}

void Timer0_Stop(void) {
    LPC_TIM0->TCR = 0x00;
}

unsigned int Timer0_GetValue(void) {
    return LPC_TIM0->TC;
}

// ============ TURBIDITY FUNCTIONS ============

void ADC_Init(void) {
    LPC_SC->PCONP |= (1 << 12);             // Power on ADC
    
    // Configure AD0.0 (P0.23) for Turbidity
    LPC_PINCON->PINSEL1 &= ~(3 << 14);
    LPC_PINCON->PINSEL1 |= (1 << 14);       // P0.23 as AD0.0
    LPC_PINCON->PINMODE1 &= ~(3 << 14);
    LPC_PINCON->PINMODE1 |= (2 << 14);      // No pull-up/pull-down
    
    // Configure AD0.1 (P0.24) for pH Sensor
    LPC_PINCON->PINSEL1 &= ~(3 << 16);
    LPC_PINCON->PINSEL1 |= (1 << 16);       // P0.24 as AD0.1
    LPC_PINCON->PINMODE1 &= ~(3 << 16);
    LPC_PINCON->PINMODE1 |= (2 << 16);      // No pull-up/pull-down
    
    // Configure AD0.2 (P0.25) for DO Sensor
    LPC_PINCON->PINSEL1 &= ~(3 << 18);
    LPC_PINCON->PINSEL1 |= (1 << 18);       // P0.25 as AD0.2
    LPC_PINCON->PINMODE1 &= ~(3 << 18);
    LPC_PINCON->PINMODE1 |= (2 << 18);      // No pull-up/pull-down
    
    LPC_ADC->ADCR = (24 << 8) | (1 << 21);  // CLKDIV=24, PDN=1
}

uint16_t ADC_Read(uint8_t channel) {
    uint32_t result;
    
    LPC_ADC->ADCR &= 0xFFFFFF00;
    LPC_ADC->ADCR |= (1 << channel) | (1 << 24);  // Select channel and start
    
    while(!(LPC_ADC->ADGDR & (1 << 31)));         // Wait for conversion
    
    result = LPC_ADC->ADGDR;
    return (result >> 4) & 0xFFF;                 // 12-bit result
}

uint16_t read_turbidity(void) {
    uint16_t adc_value;
    float voltage;
    float ntu;
    
    // Average 10 readings
    uint32_t sum = 0;
    for(uint8_t i = 0; i < 10; i++) {
        sum += ADC_Read(TURBIDITY_CHANNEL);
        delay_ms(10);
    }
    adc_value = sum / 10;
    
    // Convert to voltage
    voltage = ((float)adc_value * 3.3) / 4095.0;
    
    // Custom calibration
    if(voltage >= VOLTAGE_CLEAN) {
        ntu = 0;
    } else if(voltage >= VOLTAGE_CLEAN - 0.1) {
        ntu = (VOLTAGE_CLEAN - voltage) * 200.0;
    } else if(voltage >= VOLTAGE_CLEAN - 0.3) {
        ntu = 20.0 + ((VOLTAGE_CLEAN - 0.1) - voltage) * 400.0;
    } else if(voltage >= VOLTAGE_TURBID) {
        ntu = 100.0 + ((VOLTAGE_CLEAN - 0.3) - voltage) * 800.0;
    } else {
        ntu = 500.0 + (VOLTAGE_TURBID - voltage) * 1000.0;
    }
    
    if(ntu < 0) ntu = 0;
    return (uint16_t)ntu;
}

void display_turbidity_status(uint16_t ntu) {
    if(ntu <= TURBIDITY_GOOD) {
        uart_send_string("Status: EXCELLENT - Optimal clarity\r\n");
    } else if(ntu <= TURBIDITY_ACCEPTABLE) {
        uart_send_string("Status: GOOD - Acceptable clarity\r\n");
    } else if(ntu <= TURBIDITY_WARNING) {
        uart_send_string("Status: WARNING - Check filters\r\n");
    } else if(ntu <= TURBIDITY_CRITICAL) {
        uart_send_string("Status: POOR - Action needed\r\n");
    } else {
        uart_send_string("Status: CRITICAL - Urgent action required\r\n");
    }
}

// ============ pH SENSOR FUNCTIONS ============

float read_ph_sensor(void) {
    uint16_t adc_value;
    float voltage;
    float phValue;
    
    // Average 10 readings
    uint32_t sum = 0;
    for(uint8_t i = 0; i < 10; i++) {
        sum += ADC_Read(PH_CHANNEL);
        delay_ms(10);
    }
    adc_value = sum / 10;
    
    // Convert to voltage
    voltage = ((float)adc_value * VREF) / 4095.0f;
    
    if(voltage <= 2.8f) {
        phValue = -10.0f * (voltage - 2.6f) + 10.0f;
    } else if(voltage <= 3.23f) {
        phValue = -4.76f * (voltage - 2.8f) + 8.0f;
    } else {
        phValue = -57.14f * (voltage - 3.23f) + 6.0f;
    }
    
    if(phValue < 0.0f)  phValue = 0.0f;
    if(phValue > 14.0f) phValue = 14.0f;
    
    return phValue;
}

void display_ph_status(float phValue) {
    if(phValue >= 0 && phValue < 6) {
        uart_send_string("Status: ACIDIC\r\n");
    } else if(phValue >= 6 && phValue <= 8) {
        uart_send_string("Status: NEUTRAL\r\n");
    } else if(phValue > 8 && phValue <= 14) {
        uart_send_string("Status: ALKALINE\r\n");
    } else {
        uart_send_string("Status: OUT OF RANGE\r\n");
    }
}

// ============ DO SENSOR FUNCTIONS ============

float DO_GetVoltage(unsigned int adc_val) {
    return ((float)adc_val * VREF) / ADC_MAX;
}

float DO_ConvertToMgL(float voltage) {
    if (voltage < 0.0f) voltage = 0.0f;
    if (voltage > 3.0f) voltage = 3.0f;
    return (voltage / 3.0f) * 20.0f;
}
// ============ LCD FUNCTIONS - 8-bit Mode ============

void lcd_init(void) {
    LPC_GPIO0->FIODIR |= (RS | EN | LCD_DATA_MASK);
    
    LPC_GPIO0->FIOCLR = (RS | EN | LCD_DATA_MASK);
    
    delay_ms(50);
    
    lcd_command(0x38);
    delay_ms(5);
    lcd_command(0x38);
    delay_us(150);
    lcd_command(0x38);
    lcd_command(0x38);
    lcd_command(0x0C);
    lcd_command(0x06);
    lcd_command(0x01);
    delay_ms(2);
}

void lcd_enable_pulse(void) {
    LPC_GPIO0->FIOSET = EN;
    delay_us(10);
    LPC_GPIO0->FIOCLR = EN;
    delay_us(10);
}

void lcd_command(unsigned char cmd) {
    LPC_GPIO0->FIOCLR = RS;
    LPC_GPIO0->FIOCLR = LCD_DATA_MASK;
    LPC_GPIO0->FIOSET = ((unsigned int)cmd << LCD_DATA_SHIFT) & LCD_DATA_MASK;
    lcd_enable_pulse();
    delay_us(50);
}

void lcd_data(unsigned char data) {
    LPC_GPIO0->FIOSET = RS;
    LPC_GPIO0->FIOCLR = LCD_DATA_MASK;
    LPC_GPIO0->FIOSET = ((unsigned int)data << LCD_DATA_SHIFT) & LCD_DATA_MASK;
    lcd_enable_pulse();
    delay_us(50);
}

void lcd_string(char *str) {
    while(*str) {
        lcd_data(*str++);
    }
}

void lcd_clear(void) {
    lcd_command(0x01);
    delay_ms(2);
}

void lcd_goto(unsigned char row, unsigned char col) {
    unsigned char position;
    if(row == 0)
        position = 0x80 + col;
    else
        position = 0xC0 + col;
    lcd_command(position);
}

void lcd_float(float value, unsigned char decimals) {
    int int_part;
    float frac_part;
    unsigned int frac_int;
    unsigned char i;
    char buffer[16];
    unsigned char index = 0;
    
    if(value < 0) {
        lcd_data('-');
        value = -value;
    }
    
    int_part = (int)value;
    
    if(int_part == 0) {
        buffer[index++] = '0';
    } else {
        unsigned char temp_index = 0;
        char temp[10];
        while(int_part > 0) {
            temp[temp_index++] = (int_part % 10) + '0';
            int_part /= 10;
        }
        while(temp_index > 0) {
            buffer[index++] = temp[--temp_index];
        }
    }
    
    if(decimals > 0) {
        buffer[index++] = '.';
        frac_part = value - (int)value;
        for(i = 0; i < decimals; i++) {
            frac_part *= 10;
        }
        frac_int = (unsigned int)frac_part;
        
        for(i = 0; i < decimals; i++) {
            unsigned int divisor = 1;
            unsigned char j;
            for(j = 0; j < (decimals - i - 1); j++) {
                divisor *= 10;
            }
            buffer[index++] = (frac_int / divisor) % 10 + '0';
        }
    }
    
    buffer[index] = '\0';
    lcd_string(buffer);
}

// ============ UART0 FUNCTIONS ============

void init_uart0(void) {
    LPC_SC->PCONP |= (1<<3);
    LPC_SC->PCLKSEL0 &= ~(3<<6);
    LPC_SC->PCLKSEL0 |= (1<<6);
    
    LPC_PINCON->PINSEL0 &= ~((3<<4) | (3<<6));
    LPC_PINCON->PINSEL0 |= ((1<<4) | (1<<6));
    
    LPC_UART0->LCR = 0x83;
    uint16_t dl = 651;
    LPC_UART0->DLL = dl & 0xFF;
    LPC_UART0->DLM = (dl >> 8) & 0xFF;
    LPC_UART0->LCR = 0x03;
    LPC_UART0->FCR = 0x07;
}

void uart_send_char(char c) {
    while(!(LPC_UART0->LSR & (1<<5)));
    LPC_UART0->THR = c;
}

void uart_send_string(char *str) {
    while(*str) {
        if(*str == '\n') {
            uart_send_char('\r');
        }
        uart_send_char(*str++);
    }
}

// ============ UART3 FUNCTIONS (ESP Communication) ============

void UART3_Init(void) {
    LPC_SC->PCONP |= (1 << 25);        // Power on UART3
    LPC_PINCON->PINSEL0 |= (0xA << 0); // P0.0 as TXD3, P0.1 as RXD3
    LPC_UART3->LCR = 0x83;             // 8-bit, 1 stop bit, DLAB=1
    LPC_UART3->DLL = 0xA2;             // 9600 baud @ 25MHz PCLK
    LPC_UART3->LCR = 0x03;             // DLAB=0
}

void UART3_SendChar(char c) {
    while(!(LPC_UART3->LSR & (1 << 5))); // Wait for THR empty
    LPC_UART3->THR = c;
}

void UART3_SendString(const char *s) {
    while(*s) UART3_SendChar(*s++);
}

// ============ DELAY FUNCTIONS ============

void delay_ms(uint32_t ms) {
    for(uint32_t i = 0; i < ms; i++) {
        for(volatile uint32_t j = 0; j < 10000; j++);
    }
}

void delay_us(uint32_t us) {
    for(volatile uint32_t i = 0; i < us * 10; i++);
}
